建表：

CREATE TABLE `iom_pcc_log` (
`ID` bigint(14) NOT NULL,
`PROVINCE_CODE` varchar(20) DEFAULT NULL COMMENT '前置机的省份编码，如果是集团侧，则编码为999',
`IMSI` varchar(50) DEFAULT NULL COMMENT 'IMSI号',
`ACC_NBR` varchar(50) DEFAULT NULL COMMENT '业务号码',
`INTERFACE_TYPE` varchar(50) DEFAULT NULL COMMENT '接口类型 6个接口的接口编码如：addUserInfoSyn',
`CREATE_TIME` datetime DEFAULT NULL COMMENT '创建时间',
`SYS_CODE` varchar(50) DEFAULT NULL COMMENT '发送系统000：集中IOM;001：HSS;002：cBSS;003：省BSS;004: 全业务;005：ES;006: PCC'',',
`SEND_TIME` datetime DEFAULT NULL,
`RECEIVE_TIME` datetime DEFAULT NULL,
`SEND_FILE_INFO` varchar(500) DEFAULT NULL COMMENT '发送报文存储位置信息',
`RECEIVE_FILE_INFO` varchar(500) DEFAULT NULL COMMENT '接收报文存储位置信息',
PRIMARY KEY (`ID`),
KEY `index_acc_nbr` (`ACC_NBR`) USING BTREE,
KEY `index_interface_type` (`INTERFACE_TYPE`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT '限速表';
